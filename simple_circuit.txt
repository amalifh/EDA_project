INPUT a
INPUT b
INPUT c_in

# Gates
# Sum (S) Logic: S = (a XOR b) XOR c_in
n1 = XOR(a, b)            
n2 = XOR(n1, c_in)        

# Carry-Out (c_out) Logic: c_out = (a AND b) OR (c_in AND (a XOR b))
n3 = AND(a, b)            
n4 = AND(c_in, n1)       
n5 = OR(n3, n4)           

# Output
OUTPUT S 
OUTPUT n2
OUTPUT c_out 
OUTPUT n5
